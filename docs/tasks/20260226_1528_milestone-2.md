# Milestone 2: 表現ルールの CRUD + 有効/無効切替と DB 永続化

## 概要

// expression-rule.model.ts
import { Result } from '@/backend/contexts/shared/domain/models/result.model';
import { Timestamp } from '@/backend/contexts/shared/domain/models/timestamp.model';

type ExpressionRuleProps = {
	id: string;
	ngExpression: string;
	okExpression: string;
	description: string | null;
	enabled: boolean;
	createdAt: Timestamp;
	updatedAt: Timestamp;
};

export class ExpressionRule {
	private constructor(private readonly props: ExpressionRuleProps) {}

	static create(params: {
		ngExpression: string;
		okExpression: string;
		description?: string;
	}): Result<ExpressionRule, string> {
		if (!params.ngExpression.trim()) {
			return Result.err('NG表現は必須です');
		}
		if (!params.okExpression.trim()) {
			return Result.err('OK表現は必須です');
		}
		const now = Timestamp.now();
		return Result.ok(
			new ExpressionRule({
				id: crypto.randomUUID(),
				ngExpression: params.ngExpression.trim(),
				okExpression: params.okExpression.trim(),
				description: params.description?.trim() || null,
				enabled: true,
				createdAt: now,
				updatedAt: now,
			}),
		);
	}

	static reconstruct(props: ExpressionRuleProps): ExpressionRule {
		return new ExpressionRule(props);
	}

	get id(): string {
		return this.props.id;
	}
	get ngExpression(): string {
		return this.props.ngExpression;
	}
	get okExpression(): string {
		return this.props.okExpression;
	}
	get description(): string | null {
		return this.props.description;
	}
	get enabled(): boolean {
		return this.props.enabled;
	}
	get createdAt(): Timestamp {
		return this.props.createdAt;
	}
	get updatedAt(): Timestamp {
		return this.props.updatedAt;
	}

	update(params: {
		ngExpression: string;
		okExpression: string;
		description?: string;
	}): Result<ExpressionRule, string> {
		if (!params.ngExpression.trim()) {
			return Result.err('NG表現は必須です');
		}
		if (!params.okExpression.trim()) {
			return Result.err('OK表現は必須です');
		}
		return Result.ok(
			new ExpressionRule({
				...this.props,
				ngExpression: params.ngExpression.trim(),
				okExpression: params.okExpression.trim(),
				description: params.description?.trim() || null,
				updatedAt: Timestamp.now(),
			}),
		);
	}

	toggleEnabled(): ExpressionRule {
		return new ExpressionRule({
			...this.props,
			enabled: !this.props.enabled,
			updatedAt: Timestamp.now(),
		});
	}
}

				expect(result.value.ngExpression).toBe('NG表現');

#### import 先（既存コード）

- `@/backend/contexts/shared/domain/models/result.model` → `Result`
- `@/backend/contexts/shared/domain/models/timestamp.model` → `Timestamp`

#### 実装パターン

- **ExpressionRule モデル**: Rich Domain Model パターン。`create()` でバリデーション付き生成、`reconstruct()` で DB からの復元。`update()` と `toggleEnabled()` は新しいインスタンスを返す（イミュータブル）。
- **Repository インターフェース**: ドメインモデル型のみを引数・戻り値に使用。`save()` は upsert セマンティクス（新規 + 更新の両方に対応）。
- **UseCase**: クラスベース + コンストラクタ DI。Repository インターフェースを受け取る。Result 型の失敗を Error に変換して throw。
- 既存の `Result`, `Timestamp` モデル（Milestone 1 で作成済み）と同じパターンで実装。

#### 注意事項

- `crypto.randomUUID()` は Node.js 19+ / ブラウザで利用可能。Next.js サーバーサイドでは問題なく使える。
- テストで `vi.mocked()` を使うには vitest の `globals: true` 設定が必要（`vitest.config.ts` で設定済み）。
- domain 層は外部依存なし（`shared/domain` のみ参照可）。`@prisma/client` 等の import は禁止。

// prisma-expression-rule.repository.ts
import type { ExpressionRule as PrismaExpressionRule } from '@prisma/client';
import { prisma } from '@/backend/contexts/shared/infrastructure/db/prisma-client';
import { ExpressionRule } from '../../domain/models/expression-rule.model';
import type { ExpressionRuleRepository } from '../../domain/repositories/expression-rule.repository';
import { Timestamp } from '@/backend/contexts/shared/domain/models/timestamp.model';

export class PrismaExpressionRuleRepository implements ExpressionRuleRepository {
	async findAll(): Promise<ExpressionRule[]> {
		const rows = await prisma.expressionRule.findMany({
			orderBy: { createdAt: 'desc' },
		});
		return rows.map(this.toDomain);
	}

	async findById(id: string): Promise<ExpressionRule | null> {
		const row = await prisma.expressionRule.findUnique({ where: { id } });
		return row ? this.toDomain(row) : null;
	}

	async save(rule: ExpressionRule): Promise<void> {
		await prisma.expressionRule.upsert({
			where: { id: rule.id },
			create: {
				id: rule.id,
				ngExpression: rule.ngExpression,
				okExpression: rule.okExpression,
				description: rule.description,
				enabled: rule.enabled,
				createdAt: rule.createdAt.toDate(),
				updatedAt: rule.updatedAt.toDate(),
			},
			update: {
				ngExpression: rule.ngExpression,
				okExpression: rule.okExpression,
				description: rule.description,
				enabled: rule.enabled,
				updatedAt: rule.updatedAt.toDate(),
			},
		});
	}

	async delete(id: string): Promise<void> {
		await prisma.expressionRule.delete({ where: { id } });
	}

	private toDomain(row: PrismaExpressionRule): ExpressionRule {
		return ExpressionRule.reconstruct({
			id: row.id,
			ngExpression: row.ngExpression,
			okExpression: row.okExpression,
			description: row.description,
			enabled: row.enabled,
			createdAt: Timestamp.fromDate(row.createdAt),
			updatedAt: Timestamp.fromDate(row.updatedAt),
		});
	}
}

	trigger: React.ReactNode;
	title: string;
	defaultValues?: {
		ngExpression: string;
		okExpression: string;
		description: string;
	};

		ngExpression: string;
		okExpression: string;
		description?: string;
	}) => Promise<{ success: boolean; error?: string }>;

	trigger,
	title,
	defaultValues,
	onSubmit,
}: RuleDialogProps) {

#### 実装パターン

- **PrismaExpressionRuleRepository**: Prisma クライアントを使用した Repository 実装。`toDomain()` private メソッドで ORM 型 → ドメインモデル変換。`save()` は `upsert` で新規作成・更新を統一的に処理。Milestone 1 の `prisma-client.ts` シングルトンを利用。
- **Composition Root**: 各 UseCase のファクトリ関数をエクスポート。毎回新しい Repository インスタンスを生成（Prisma クライアント自体はシングルトンなので問題ない）。将来の Milestone でナレッジ・ソース関連の UseCase もここに追加する。
- **Loader**: BFF パターン。ドメインモデルからフロントエンド用の plain object（`RuleListItem`）に変換して返す。
- **Actions**: `'use server'` ディレクティブ付き。try-catch で UseCase の例外をキャッチし `{ success, error }` 形式で返す。成功時に `revalidatePath('/rules')` で Server Component を再検証。
- **RuleDialog**: 汎用的な追加/編集ダイアログ。`onSubmit` コールバックで作成・更新の両方に対応。`defaultValues` の有無で追加/編集モードを切り替え。
- **RulesTable**: Client Component。テーブル表示 + Switch による有効/無効切替 + 編集ダイアログ + 削除確認ダイアログ。Server Actions を直接 import して呼び出す。
- **rules/page.tsx**: Server Component。`loadRuleList()` でデータ取得し、Client Components に props として渡す。ルールが0件の場合は Empty State を表示。

#### 注意事項

- **`revalidatePath` の効果**: Server Action 内で `revalidatePath('/rules')` を呼ぶと、Next.js が `/rules` ページの Server Component を再実行し、最新データでクライアントを更新する。Client Component 側で手動リフレッシュは不要。
- **Dialog の `DialogDescription`**: shadcn/ui の Dialog は `DialogDescription` がないとアクセシビリティ警告が出る。削除確認ダイアログでは `DialogDescription` を含めること。RuleDialog（追加/編集）では不要な場合は `aria-describedby={undefined}` を DialogContent に設定してもよい。
- **rules/page.tsx は Server Component のまま**: `RuleDialog` と `RulesTable` が Client Component なので、ページ自体は Server Component として維持できる。ただし `RuleDialog` を直接ページ内でレンダリングしているため、Next.js がクライアントバンドルに含める。`createRule` action の import は Server Action なので問題ない。
- **`?highlight={ruleId}` のハイライト**: Next.js 15 では `searchParams` は Promise。Server Component 内で `await` して `highlight` パラメータを取得し、`RulesTable` に渡す。
- **Prisma `upsert` の `updatedAt`**: Prisma スキーマで `@updatedAt` 属性を付けているが、`upsert` の `update` で明示的に `updatedAt` を指定する。ドメインモデル内で管理した Timestamp を優先するため。
